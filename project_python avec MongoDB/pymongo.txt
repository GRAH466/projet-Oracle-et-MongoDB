from pymongo import MongoClient
import pymongo

// se connecter a la connexion de mongo db
GRAH=MongoClient(host='localhost',port=27017)

//se connecter a la base de donnée
db=GRAH.etd

//inserer un dictionnaire dans la collection de mongo db
yabi= {
    'nom':'vice versa',
   
    'annee':2022,
       'acteurs':'sidibe issa',
    'realisateurs':{ 
    'nom':'grah','prenom':'lowa-ezechiel'
    }}

print(result)

result=db.Cinema.find()

//afficher les 5 premières lignes de la collection
result=db.Cinema.find().limit(5)
print(result)


//
result=db.Cinema.find().sort("annee",pymongo.ASCENDING)
for a in result:
